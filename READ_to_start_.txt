# Windows Quickstart
python -m venv .venv

Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
.\.venv\Scripts\Activate.ps1

pip install -U pip -r requirements.txt

Copy-Item .env.example .env  # add your API key to .env
notepad .env                  # set LLM_API_KEY (and optionally LLM_MODEL)

# Analyze a single email (convert .eml to JSON first if needed)
python -m app.eml_to_json "C:\path\to\email.eml" ".\out\email.json"
python .\main.py ".\out\email.json"

# Analyze JSONL (all lines or first N lines); results append to out\results.jsonl
python .\main.py data\samples.jsonl
python .\main.py data\samples.jsonl 25
